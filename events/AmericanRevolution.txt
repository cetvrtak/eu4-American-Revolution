# American quest for independence
country_event = {
	id = 1021
	title = "EVTNAME1021"
	desc = "EVTDESC1021"
	picture = ANGRY_MOB_eventPicture
	
	trigger = {
		NOT = { has_global_flag = american_independence_war_started }
		NOT = { has_country_flag = american_revolution }
		capital_scope = { continent = europe }
		current_age = age_of_revolutions
		any_subject_country = {
			is_colonial_nation_of = ROOT
			liberty_desire = 50
			capital_scope = {
				OR = {
					continent = north_america
					continent = south_america
				}
			}
			if = {
				limit = { has_dlc = "Emperor" }
				any_country = {
					is_rival = ROOT
					OR = {
						alliance_with = PREV
						NOT = {
							war_with = ROOT
							truce_with = ROOT
							NOT = { has_opinion = { who = PREV value = 0 } }
						}
					}
				}
			}
		}
	}

	immediate = {
		random_subject_country = {
			limit = {
				is_colonial_nation_of = ROOT
				liberty_desire = 50
				capital_scope = {
					OR = {
						continent = north_america
						continent = south_america
					}
				}
			}
			save_global_event_target_as = revolutionary_america
		}
	}
	
	mean_time_to_happen = {
		months = 120

		modifier = {
			factor = 0.9
			economic_ideas = 6
		}
		modifier = {
			factor = 1.1
			NOT = { economic_ideas = 4 }
		}
		modifier = {
			factor = 1.1
			NOT = { economic_ideas = 2 }
		}
		modifier = {
			factor = 0.9
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.9
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 0.9
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 1.1
			stability = 1
		}
		modifier = {
			factor = 1.1
			stability = 2
		}
		modifier = {
			factor = 1.1
			stability = 3
		}
		modifier = {
			factor = 0.95
			NOT = { adm = 2 }
		}
		modifier = {
			factor = 0.85
			NOT = { adm = 1 }
		}
		modifier = {
			factor = 0.95
			is_year = 1760
		}
		modifier = {
			factor = 0.85
			is_year = 1780
		}
	}
	
	option = {
		name = "EVTOPTA1021"
		set_country_flag = american_revolution
		if = {
			limit = { has_dlc = "Emperor" }
			custom_tooltip = ally_joining_revolter_tooltip
		}
	}
}

# Treaty of Alliance
country_event = {
	id = 1024
	title = "EVTNAME1024_a"
	desc = "EVTDESC1024_a"
	picture = DIPLOMACY_eventPicture

	trigger = {
		has_dlc = "Emperor"
		is_at_war = no
		any_country = {
			has_country_flag = american_revolution
			war_with = event_target:revolutionary_america
			is_enemy = ROOT
			NOT = { truce_with = ROOT }
		}
		has_opinion = {
			who = event_target:revolutionary_america
			value = 0
		}
		NOT = { alliance_with = event_target:revolutionary_america }
		NOT = { has_country_flag = refused_americans}
	}

	mean_time_to_happen = {
		months = 24

		modifier = {
			factor = 0.5
			had_global_flag = { flag = american_independence_war_started days = 730 }
		}
		modifier = {
			factor = 0.5
			had_global_flag = { flag = american_independence_war_started days = 1095 }
		}
	}

	option = {
		name = "EVTOPTA1024_a"	# Anything to weaken our rival
		ai_chance = { factor = 100 }
		create_alliance = event_target:revolutionary_america
		join_all_offensive_wars_of = event_target:revolutionary_america
		set_country_flag = joined_americans
	}

	option = {
		name = "EVTOPTB1024_a"	# This is not a good time
		ai_chance = { factor = 1 }
		add_prestige = -5
		set_country_flag = refused_americans
	}
}

		any_ally = {

			}
		}

			has_country_flag = american_revolution
			war_with = event_target:revolutionary_america

		}
